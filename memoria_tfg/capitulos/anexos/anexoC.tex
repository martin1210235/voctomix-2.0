\Large{\textsc{\textcolor{nar}{C1. Introducción}}}
\vspace{0.2cm}

\normalsize{Este anexo reúne en un solo lugar los ficheros principales del proyecto, para
consultarlos sin necesidad de acceder al repositorio. C2 compara la estructura de directorios
antes y después del desarrollo; C3 muestra la interfaz gráfica original; C4 recoge los

Capítulo~\ref{chap:pruebas}.}

\normalsize{El \texttt{Dockerfile} construye la imagen base instalando GStreamer, FFmpeg,
Python y la librería \texttt{pika}. El \texttt{docker-compose.yml} arranca los once
servicios del sistema. La clave de la arquitectura es \texttt{network\_mode:
"service:voctocore"}: con esta opción todos los contenedores de cámara comparten la
red de voctocore y se comunican por \texttt{localhost}, sin modificar ninguna línea
del código original.}
\vspace{0.3cm}

\noindent\textbf{\small Dockerfile}
\vspace{0.1cm}

\begin{lstlisting}[style=codefile, language=dockerfile]
FROM ubuntu:22.04

ENV DEBIAN_FRONTEND=noninteractive

RUN apt-get update && apt-get install -y \
    python3 \
    python3-pip \
    python3-gi \
    python3-gi-cairo \
    python3-scipy \
    python3-numpy \
    gir1.2-gstreamer-1.0 \
    gir1.2-gst-plugins-base-1.0 \
    gir1.2-gtk-3.0 \
    gstreamer1.0-tools \
    gstreamer1.0-plugins-base \
    gstreamer1.0-plugins-good \
    gstreamer1.0-plugins-bad \
    gstreamer1.0-plugins-ugly \
    gstreamer1.0-libav \
    ffmpeg \
    netcat-openbsd \
    iproute2 \
    && pip3 install --no-cache-dir pika==1.3.2 \
    && apt-get clean \
    && rm -rf /var/lib/apt/lists/*

WORKDIR /opt/voctomix

COPY . /opt/voctomix/

HEALTHCHECK --interval=5s --timeout=3s --retries=12 \
    --start-period=20s \
    CMD bash -c "echo '' | nc -zw1 localhost ${VOCTOCORE_PORT:-9999}" || exit 1
\end{lstlisting}

\noindent\textbf{\small docker-compose.yml}
\vspace{0.1cm}

\begin{lstlisting}[style=codefile, morecomment={[l]{\#}}]
x-video-env: &video-env
  WIDTH:     ${WIDTH:-1920}
  HEIGHT:    ${HEIGHT:-1080}
  FRAMERATE: ${FRAMERATE:-25}
  AUDIORATE: ${AUDIORATE:-48000}

# Dedicated bridge network: isolates the telemetry service from the
# voctocore network stack, removing the single point of failure that
# would take down all services if voctocore crashed.
networks:
  voctomix_net:
    driver: bridge

services:
  # 0. RABBITMQ (Message Broker - event decoupling)
  rabbitmq:
    image: rabbitmq:3-management
    container_name: rabbitmq
    networks:
      - voctomix_net
    ports:
      - "5672:5672"
      - "15672:15672"
    environment:
      RABBITMQ_DEFAULT_USER: ${RABBITMQ_USER:-voctomix}
      RABBITMQ_DEFAULT_PASS: ${RABBITMQ_PASS:-voctomix123}
    healthcheck:
      test: ["CMD", "rabbitmq-diagnostics", "ping"]
      interval: 10s
      timeout: 5s
      retries: 6
      start_period: 30s

  # 1. VOCTOCORE (Central Server)
  voctocore:
    build: .
    container_name: voctocore
    command: python3 voctocore/voctocore.py
    volumes:
      - ./:/opt/voctomix
      - ./images:/opt/images
    working_dir: /opt/voctomix
    networks:
      - voctomix_net
    ports:
      - "9998:9998"
      - "9998:9998/udp"
      - "9999:9999"
      - "10000:10000"
      - "10001:10001"
      - "10002:10002"
      - "10003:10003"
      - "10004:10004"
      - "10005:10005"
      - "11000:11000"
      - "12000:12000"
      - "13000-13005:13000-13005"
      - "14000-14005:14000-14005"
      - "15000:15000"
    healthcheck:
      test: ["CMD", "bash", "-c", "echo '' | nc -zw1 localhost 9999"]
      interval: 5s
      timeout: 3s
      retries: 12
      start_period: 20s

  # 2. TELEMETRY SERVICE (Telemetry and JSON logging)
  telemetry:
    image: voctomix-voctocore:latest
    container_name: telemetry
    command: python3 example-scripts/ffmpeg/telemetry_service.py
    environment:
      - TELEMETRY_USE_UDP=1
      - HOST_IP=${HOST_IP}
      - VOCTOCORE_HOST=voctocore
      - RABBITMQ_HOST=rabbitmq
      - RABBITMQ_USER=${RABBITMQ_USER:-voctomix}
      - RABBITMQ_PASS=${RABBITMQ_PASS:-voctomix123}
      - SAVE_LOGS=${SAVE_LOGS:-false}
    user: "${COMPOSE_UID:-1000}:${COMPOSE_GID:-1000}"
    volumes:
      - ./:/opt/voctomix
    working_dir: /opt/voctomix
    networks:
      - voctomix_net
    ports:
      - "8080:8080"
    depends_on:
      voctocore:
        condition: service_healthy
      rabbitmq:
        condition: service_healthy

  # 3. CAMERA 1
  cam1:
    build: .
    container_name: cam1
    environment:
      <<: *video-env
      CAM1_SOURCE: ${CAM1_SOURCE:-https://...}
      CAM1_PORT:   ${CAM1_PORT:-10000}
    command:
      - sh
      - -c
      - >-
        ffmpeg -y -nostdin
        -i "$$CAM1_SOURCE"
        -ac 2
        -filter_complex "[0:v] scale=$$WIDTH:$$HEIGHT,fps=$$FRAMERATE [v];
                         [0:a] aresample=$$AUDIORATE [a]"
        -map "[v]" -map "[a]"
        -pix_fmt yuv420p -c:v rawvideo -c:a pcm_s16le
        -f matroska tcp://localhost:$$CAM1_PORT
    volumes:
      - ./:/opt/voctomix
    working_dir: /opt/voctomix
    network_mode: "service:voctocore"
    restart: on-failure
    depends_on:
      voctocore:
        condition: service_healthy

  # cam2, cam3: identicos a cam1 con sus respectivos puertos y fuentes.

  # 4. CAMERA 4 (con filtro de colorimetria BT.709 explicito)
  cam4:
    build: .
    container_name: cam4
    environment:
      <<: *video-env
      CAM4_SOURCE: ${CAM4_SOURCE:-https://...}
      CAM4_PORT:   ${CAM4_PORT:-10003}
    command:
      - sh
      - -c
      - >-
        ffmpeg -y -nostdin -stream_loop -1
        -i "$$CAM4_SOURCE"
        -filter_complex "[0:v] format=yuv420p,
          scale=$$WIDTH:$$HEIGHT:out_range=tv,fps=$$FRAMERATE,
          setparams=color_primaries=bt709:color_trc=bt709:colorspace=bt709 [v];
          [0:a] aresample=$$AUDIORATE [a]"
        -map "[v]" -map "[a]"
        -color_primaries bt709 -color_trc bt709 -colorspace bt709 -color_range tv
        -c:v rawvideo -c:a pcm_s16le
        -f matroska tcp://localhost:$$CAM4_PORT
    volumes:
      - ./:/opt/voctomix
    working_dir: /opt/voctomix
    network_mode: "service:voctocore"
    restart: on-failure
    depends_on:
      voctocore:
        condition: service_healthy

  # 5. BREAK (video de pausa con audio independiente)
  break:
    build: .
    container_name: break
    environment:
      <<: *video-env
      BREAK_VIDEO: ${BREAK_VIDEO:-/opt/voctomix/videos/video_pausa.mp4}
      BREAK_AUDIO: ${BREAK_AUDIO:-/opt/voctomix/videos/musica_pausa.mp3}
      BREAK_PORT:  ${BREAK_PORT:-10004}
    command:
      - sh
      - -c
      - >-
        ffmpeg -y -nostdin -stream_loop -1 -i "$$BREAK_VIDEO"
        -re -stream_loop -1 -i "$$BREAK_AUDIO"
        -filter_complex "[0:v] format=yuv420p,scale=$$WIDTH:$$HEIGHT:out_range=tv,
          fps=$$FRAMERATE [v]; [1:a] aresample=$$AUDIORATE,pan=stereo|c0=c0|c1=c1 [a]"
        -map "[v]" -map "[a]"
        -c:v rawvideo -c:a pcm_s16le
        -f matroska tcp://localhost:$$BREAK_PORT
    network_mode: "service:voctocore"
    restart: on-failure
    depends_on:
      voctocore:
        condition: service_healthy

  # 6. INTRO (bucle con pista de silencio generada en tiempo real)
  intro:
    build: .
    container_name: intro
    environment:
      <<: *video-env
      INTRO_VIDEO: ${INTRO_VIDEO:-/opt/voctomix/videos/intro.mp4}
      INTRO_PORT:  ${INTRO_PORT:-10005}
    command:
      - bash
      - -c
      - |
        while true; do
          ffmpeg -loglevel quiet -nostats -re \
            -i "$$INTRO_VIDEO" \
            -f lavfi -i anullsrc=r=$$AUDIORATE:cl=stereo \
            -pix_fmt yuv420p -s "$${WIDTH}x$${HEIGHT}" -r "$$FRAMERATE" \
            -c:v rawvideo -c:a pcm_s16le \
            -map 0:v:0 -map 1:a:0 -shortest \

\begin{lstlisting}[style=codefile, language=python3]
#!/usr/bin/env python3

import socket
import time
import argparse
import json
import numpy as np
from datetime import datetime

MODES = ["fs", "sbs", "pip", "lec"]
SRC_A = "cam1"
SRC_B = "cam2"


def send_recv(sock, cmd, expected_prefix, timeout=3.0):
    sock.sendall((cmd + "\n").encode())
    buf = ""
    deadline = time.monotonic() + timeout
    while time.monotonic() < deadline:
        try:
            chunk = sock.recv(4096).decode(errors="replace")
            buf += chunk
            for line in buf.split("\n"):
                line = line.strip()
                if line.startswith(expected_prefix):
                    return line
        except socket.timeout:
            continue
    return None


def measure(host, port, n_measurements):
    results = []
    sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    sock.settimeout(0.5)
    sock.connect((host, port))

    for i in range(n_measurements):
        mode = MODES[i % len(MODES)]
        cmd = f"transition {mode}({SRC_A},{SRC_B})"
        expected = f"composite {mode}"

        t0 = time.monotonic()
        response = send_recv(sock, cmd, expected)
        t1 = time.monotonic()

        if response is None:
            continue

        ms = (t1 - t0) * 1000
        results.append({"mode": mode, "latency_ms": round(ms, 3)})
        time.sleep(0.15)

    sock.close()
    return results


def stats(results):
    if not results:
        return {}
    vals = [r["latency_ms"] for r in results]
    return {
        "n": len(vals),
        "min_ms":    round(float(np.min(vals)), 2),
        "median_ms": round(float(np.median(vals)), 2),
        "mean_ms":   round(float(np.mean(vals)), 2),
        "max_ms":    round(float(np.max(vals)), 2),
        "std_ms":    round(float(np.std(vals)), 2),
        "p95_ms":    round(float(np.percentile(vals, 95)), 2),
    }


def histogram(results, bins=None):
    if bins is None:
        bins = [1.0, 1.5, 2.0, 2.5, 3.0, 3.5, 4.0, 4.5, 5.0, 6.0, 8.0]
    vals = [r["latency_ms"] for r in results]
    hist = {}
    for lo, hi in zip(bins[:-1], bins[1:]):
        hist[f"{lo}-{hi}"] = sum(1 for v in vals if lo <= v < hi)
    return hist


def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--host",  default="localhost")
    parser.add_argument("--port",  type=int, default=9999)
    parser.add_argument("--n",     type=int, default=32)
    parser.add_argument("--label", default="docker")
    args = parser.parse_args()

    results = measure(args.host, args.port, args.n)
    s = stats(results)
    h = histogram(results)

    output = {
        "label":     args.label,
        "timestamp": datetime.now().isoformat(),
        "host":      args.host,
        "port":      args.port,
        "stats":     s,
        "histogram": h,
        "raw":       results,
    }
    out_path = f"sessions/composite_latency_{args.label}.json"
    with open(out_path, "w") as f:
        json.dump(output, f, indent=2)


if __name__ == "__main__":
    main()

#!/usr/bin/env python3

import subprocess
import time
import json
import argparse
import numpy as np
from datetime import datetime


def container_is_healthy(name):
    out = subprocess.run(
        ["docker", "inspect", "--format",
         "{{.State.Status}} {{.State.Health.Status}}", name],
        capture_output=True, text=True
    ).stdout.strip()
    return out == "running healthy"


def measure_recovery(container, n):
    results = []

    for i in range(n):
        # Kill PID 1 inside the container to simulate a real crash.
        # docker stop would send SIGTERM and skip the restart policy.
        subprocess.run(["docker", "exec", container, "kill", "-9", "1"],
                       capture_output=True)
        t_stop = time.monotonic()

        deadline = t_stop + 60
        recovered = False
        while time.monotonic() < deadline:
            time.sleep(0.5)
            if container_is_healthy(container):
                t_healthy = time.monotonic()
                recovered = True
                break

        if not recovered:
            continue

        ms = round((t_healthy - t_stop) * 1000)
        results.append(ms)
        time.sleep(2)

    return results


def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--container", default="cam1")
    parser.add_argument("--n", type=int, default=10)
    args = parser.parse_args()

    results = measure_recovery(args.container, args.n)

    if not results:
        return

    vals = np.array(results)
    output = {
        "container": args.container,
        "timestamp": datetime.now().isoformat(),
        "stats": {
            "n":         int(len(vals)),
            "min_ms":    int(np.min(vals)),
            "median_ms": float(np.median(vals)),
            "mean_ms":   float(np.mean(vals)),
            "max_ms":    int(np.max(vals)),
            "p95_ms":    float(np.percentile(vals, 95)),
        },
        "raw_ms": list(map(int, vals)),
    }
    out_path = f"sessions/resilience_{args.container}.json"
    with open(out_path, "w") as f:
        json.dump(output, f, indent=2)


if __name__ == "__main__":
    main()

                    ts = datetime.fromisoformat(ts_str)
                    if t0 is None:
                        t0 = ts
                    elapsed_min = (ts - t0).total_seconds() / 60.0
                    if max_minutes is not None and elapsed_min > max_minutes:
                        break
                    timestamps.append(ts)
                    cpu.append(float(cpu_val))
                    ram.append(float(ram_val))
                except ValueError:
                    continue
    return timestamps, cpu, ram


def to_minutes(timestamps):
    if not timestamps:
        return []
    t0 = timestamps[0]
    return [(t - t0).total_seconds() / 60.0 for t in timestamps]
