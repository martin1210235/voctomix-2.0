% ── C1. Introducción ────────────────────────────────────────────────
\Large{\textsc{\textcolor{nar}{C1. Introducción}}}
\vspace{0.2cm}

\normalsize{El punto de partida del proyecto fue el repositorio oficial de Voctomix~\cite{voctomix_github},
alojado en GitHub bajo la organización \texttt{voc}, con una historia de \textit{commits}
que se remonta a 2016. Este anexo documenta la evolución del repositorio a lo largo del
trabajo: la sección~C2 compara la estructura de directorios antes y después del desarrollo,
la sección~C3 detalla los módulos incorporados, la sección~C4 muestra la evolución de
la interfaz gráfica, la sección~C5 reproduce los ficheros de configuración INI del
sistema, la sección~C6 los ficheros de infraestructura Docker y la
sección~C7 los manifiestos de despliegue en Kubernetes.}
\vspace{0.4cm}

\normalsize{El despliegue en contenedores se define en dos ficheros.
El \texttt{Dockerfile} parte de Ubuntu~22.04~\ac{LTS} e instala en una sola capa
todas las dependencias del sistema: GStreamer~1.0 con sus \textit{plugins}, Python~3,
FFmpeg y la librería \texttt{pika} para la integración con RabbitMQ.
El fichero \texttt{docker-compose.yml} declara los diez servicios del \textit{stack}:
voctocore, RabbitMQ, el servicio de telemetría, cuatro contenedores de cámara
(\texttt{cam1}--\texttt{cam4}), el \textit{stream blanker}, el intro y el gestor
de audio, con sus relaciones de dependencia, puertos expuestos y configuración de red.}
\vspace{0.3cm}

\noindent\textbf{\small Dockerfile}
\vspace{0.1cm}

\begin{lstlisting}[style=codefile, language=dockerfile]
FROM ubuntu:22.04

ENV DEBIAN_FRONTEND=noninteractive

RUN apt-get update && apt-get install -y \
    python3 \
    python3-pip \
    python3-gi \
    python3-gi-cairo \
    python3-scipy \
    python3-numpy \
    gir1.2-gstreamer-1.0 \
    gir1.2-gst-plugins-base-1.0 \
    gir1.2-gtk-3.0 \
    gstreamer1.0-tools \
    gstreamer1.0-plugins-base \
    gstreamer1.0-plugins-good \
    gstreamer1.0-plugins-bad \
    gstreamer1.0-plugins-ugly \
    gstreamer1.0-libav \
    ffmpeg \
    netcat-openbsd \
    iproute2 \
    && pip3 install --no-cache-dir pika==1.3.2 \
    && apt-get clean \
    && rm -rf /var/lib/apt/lists/*

WORKDIR /opt/voctomix

COPY . /opt/voctomix/

HEALTHCHECK --interval=5s --timeout=3s --retries=12 \
    --start-period=20s \
    CMD bash -c "echo '' | nc -zw1 localhost ${VOCTOCORE_PORT:-9999}" || exit 1
\end{lstlisting}

        -ac 2
        -filter_complex "[0:v] scale=$$WIDTH:$$HEIGHT,fps=$$FRAMERATE [v];
                         [0:a] aresample=$$AUDIORATE [a]"
        -map "[v]" -map "[a]"
        -pix_fmt yuv420p -c:v rawvideo -c:a pcm_s16le
        -f matroska tcp://localhost:$$CAM1_PORT
    volumes:
      - ./:/opt/voctomix
    working_dir: /opt/voctomix
    network_mode: "service:voctocore"
    restart: on-failure
    depends_on:
      voctocore:
        condition: service_healthy

  # cam2, cam3: identicos a cam1 con sus respectivos puertos y fuentes.

  # 4. CAMERA 4 (con filtro de colorimetria BT.709 explicito)

    build: .
    container_name: cam4
    environment:

      CAM4_SOURCE: ${CAM4_SOURCE:-https://...}

      - >-
        ffmpeg -y -nostdin -stream_loop -1

      - ./:/opt/voctomix
    working_dir: /opt/voctomix

      - -c
      - >-

        -re -stream_loop -1 -i "$$BREAK_AUDIO"

          fps=$$FRAMERATE [v]; [1:a] aresample=$$AUDIORATE,pan=stereo|c0=c0|c1=c1 [a]"

        -f matroska tcp://localhost:$$BREAK_PORT
    network_mode: "service:voctocore"
    restart: on-failure
    depends_on:
      voctocore:
        condition: service_healthy

  # 6. INTRO (bucle con pista de silencio generada en tiempo real)
  intro:
    build: .
    container_name: intro
    environment:
      <<: *video-env
      INTRO_VIDEO: ${INTRO_VIDEO:-/opt/voctomix/videos/intro.mp4}
      INTRO_PORT:  ${INTRO_PORT:-10005}
    command:
      - bash
      - -c
      - |
        while true; do
          ffmpeg -loglevel quiet -nostats -re \
            -i "$$INTRO_VIDEO" \
            -f lavfi -i anullsrc=r=$$AUDIORATE:cl=stereo \
            -pix_fmt yuv420p -s "$${WIDTH}x$${HEIGHT}" -r "$$FRAMERATE" \
            -c:v rawvideo -c:a pcm_s16le \
            -map 0:v:0 -map 1:a:0 -shortest \
            -f matroska tcp://127.0.0.1:$$INTRO_PORT 2>/dev/null
          sleep 1
        done
    network_mode: "service:voctocore"
    restart: on-failure
    depends_on:
      voctocore:
        condition: service_healthy

  # 7. STREAM BLANKER (PAUSE + NOSTREAM en paralelo)
  stream_blanker:
    build: .
    container_name: stream_blanker
    environment:
      <<: *video-env
      PAUSE_VIDEO:          ${PAUSE_VIDEO:-/opt/voctomix/videos/starting_soon.mp4}

      BLANKER_PAUSE_PORT:   ${BLANKER_PAUSE_PORT:-17000}
      BLANKER_OFFLINE_PORT: ${BLANKER_OFFLINE_PORT:-17001}

      - bash
      - -c
      - |
        ffmpeg -nostats -loglevel warning -stream_loop -1 -re \
          -i "$$PAUSE_VIDEO" \
          -pix_fmt yuv420p -s "$${WIDTH}x$${HEIGHT}" -r "$$FRAMERATE" \
          -c:v rawvideo -an -f matroska tcp://127.0.0.1:$$BLANKER_PAUSE_PORT &
        ffmpeg -nostats -loglevel warning -stream_loop -1 -re \
          -i "$$OFFLINE_VIDEO" \
          -pix_fmt yuv420p -s "$${WIDTH}x$${HEIGHT}" -r "$$FRAMERATE" \
          -c:v rawvideo -an -f matroska tcp://127.0.0.1:$$BLANKER_OFFLINE_PORT &
        wait
    network_mode: "service:voctocore"
    restart: on-failure
    depends_on:
      voctocore:
        condition: service_healthy

  # 8. DYNAMIC AUDIO CONTROLLER (musica de fondo continua)
  audio_manager:
    build: .
    container_name: audio_manager
    environment:
      AUDIORATE:        ${AUDIORATE:-48000}
      BACKGROUND_MUSIC: ${BACKGROUND_MUSIC:-/opt/voctomix/videos/musica_pausa.mp3}
      AUDIO_PORT:       ${AUDIO_PORT:-18000}

      - sh
      - -c
      - >-
        ffmpeg -nostats -loglevel warning -re -stream_loop -1
        -i "$$BACKGROUND_MUSIC"
        -c:a pcm_s16le -ar $$AUDIORATE -ac 2 -vn
        -f matroska tcp://127.0.0.1:$$AUDIO_PORT
    network_mode: "service:voctocore"
    restart: on-failure
    depends_on:
      voctocore:
        condition: service_healthy

% ── C7. Manifiestos Kubernetes ───────────────────────────────────────
\Large{\textsc{\textcolor{nar}{C7. Manifiestos de despliegue en Kubernetes}}}
\vspace{0.2cm}

\normalsize{El despliegue en Kubernetes se define en cinco ficheros YAML.
\texttt{configmap.yaml} centraliza todos los parámetros del sistema como variables
de entorno. \texttt{pvc.yaml} reserva almacenamiento persistente para los registros
de telemetría. \texttt{rabbitmq.yaml} despliega el broker como pod independiente con
su Service interno. \texttt{studio.yaml} define el pod principal con todos los
contenedores de la aplicación, un \textit{initContainer} que espera a RabbitMQ, una
\textit{readinessProbe} sobre el puerto~9999 de voctocore y un Service de tipo
NodePort para exponer los puertos al exterior. Las credenciales se almacenan en un
\texttt{Secret} referenciado por los manifiestos pero no reproducido aquí por seguridad;
el fichero \texttt{secret.yaml.example} del repositorio sirve de plantilla.}
\vspace{0.3cm}
