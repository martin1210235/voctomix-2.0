\section{Contexto y motivaci\'on}

\begin{frame}{Contexto y motivaci\'on}
  \begin{itemize}
    \item La producci\'on audiovisual en directo depende de mezcladores
          hardware de coste muy elevado.
    \item Necesidad de personal especializado y dependencia de fabricante.
    \item Las alternativas software (OBS, vMix) son monol\'iticas y de un
          solo operador.
    \item \alert{Oportunidad}: democratizar la producci\'on broadcast con
          software libre, contenedores y control programable.
  \end{itemize}
\end{frame}

%==============================================================
\section{Objetivos}

\begin{frame}{Objetivos del trabajo}
  \begin{itemize}
    \item Extender Voctomix con capacidades de producci\'on profesional.
    \item Contenerizar el sistema completo para despliegue reproducible.
    \item A\~nadir telemetr\'ia en tiempo real para monitorizaci\'on.
    \item Validar el sistema en m\'ultiples escenarios de despliegue.
  \end{itemize}
\end{frame}

%==============================================================
\section{Estado del arte}

\begin{frame}{Estado del arte}
  \begin{itemize}
    \item Mezcladores hardware (Blackmagic ATEM): fiables pero costosos.
    \item Software monol\'itico (OBS Studio, vMix): sin API ni modo headless.
    \item Modelo REMI: producci\'on remota sobre IP.
    \item Voctomix v1.3: arquitectura cliente--servidor; desplegado como
          \emph{Production Core} en un evento deportivo real.
  \end{itemize}
\end{frame}

\section{Contribuciones}

\begin{frame}{Contribuci\'on 1 \textemdash{} Sistema de overlays}
  \begin{itemize}
    \item Overlay PNG y dos capas de texto independientes.
    \item Fundido configurable (fade-in/fade-out).
    \item Desactivaci\'on autom\'atica en cada corte de fuente.
  \end{itemize}
\end{frame}

\begin{frame}{Contribuci\'on 2 \textemdash{} Stream Blanker}
  \begin{itemize}
    \item Tres estados de salida: LIVE / PAUSE / NOSTREAM.
    \item Control de la se\~nal de programa independiente de la mezcla interna.
    \item Transiciones limpias en frontera de buffer GStreamer.
  \end{itemize}
\end{frame}

\begin{frame}{Contribuci\'on 3 \textemdash{} Audio Follows Video}
  \begin{itemize}
    \item Fundido cruzado autom\'atico de audio al cambiar de fuente.
    \item Reduce la carga cognitiva del operador en directo.
    \item Independiente del modo de composici\'on.
  \end{itemize}
\end{frame}

\begin{frame}{Contribuci\'on 4 \textemdash{} Telemetr\'ia RabbitMQ}
  \begin{itemize}
    \item Exporta el estado completo del mezclador como JSON.
    \item Eventos CHANGE (inmediatos) y STATE (peri\'odicos) v\'ia AMQP.
    \item Sin modificar el n\'ucleo de la aplicaci\'on.
  \end{itemize}
\end{frame}

\section{Despliegue}

\begin{frame}{Despliegue con Docker Compose}
  \begin{itemize}
    \item Stack de 10 servicios desplegable con un \'unico comando.
    \item \texttt{network\_mode: service:voctocore}: comunicaci\'on por loopback.
    \item Reproducibilidad total del entorno de ejecuci\'on.
  \end{itemize}
\end{frame}

\begin{frame}{Despliegue en Kubernetes}
  \begin{itemize}
    \item Pod multi-contenedor con \texttt{voctocore} y c\'amaras sidecar.
    \item RabbitMQ como StatefulSet con almacenamiento persistente.
    \item Orquestaci\'on y auto-recuperaci\'on declarativa.
  \end{itemize}
\end{frame}
